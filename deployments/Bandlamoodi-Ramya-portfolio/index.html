${p.name}

${p.desc}

View Project `; projDiv.appendChild(card); }); // Education const eduDiv = document.getElementById("eduTimeline"); PROFILE.education.forEach((e, i) => { const item = document.createElement("div"); item.className = "mb-6"; item.setAttribute("data-aos", "fade-right"); item.setAttribute("data-aos-delay", i*150); item.innerHTML = `

${e.degree}

${e.school} • ${e.year}

${e.details}

`; eduDiv.appendChild(item); }); } applyTheme(); loadData();
